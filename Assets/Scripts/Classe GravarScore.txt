using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Runtime.Serialization.Formatters.Binary;
using System.IO;

namespace Space
{
    public static class GravarScore
    {
        #region Metodo Load
        public static Type Carregar<Type>(string arquivo)
        {
            try
            {
                BinaryFormatter binaryFormater = new BinaryFormatter();
                using (FileStream fileStream = new FileStream(arquivo, FileMode.OpenOrCreate))
                {
                    return (Type)binaryFormater.Deserialize(fileStream);
                }
            }
            catch
            {
                object nulo = null;
                return (Type)nulo;
            }
        }
        #endregion

        #region Metodo Salvar

        public static void Salvar<Type>(string arquivo, Type classe)
        {
            if (File.Exists(arquivo))
            { File.Delete(arquivo); }

            BinaryFormatter binaryFormater = new BinaryFormatter();
            using (FileStream fileStream = new FileStream(arquivo, FileMode.OpenOrCreate))
            {
                binaryFormater.Serialize(fileStream, classe);
            }
        }

        #endregion

    }
}
