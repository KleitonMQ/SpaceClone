using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace Space
{
    public class InformacoesFases
    {
        private Random sorteio = new Random();
        private string[] ListaInimigo = { @"Imagens\01", @"Imagens\02", @"Imagens\03" };
        private Random sorteioCorInimigo = new Random();
        private int[] listaCores = { 0, 1, 2, 3, 4, 5 };
        public string SortearInimigo()
        {
            string inimigo = ListaInimigo[sorteio.Next(ListaInimigo.Length)];
            return inimigo;
        }
        public int SortearCor()
        {
            int cor = listaCores[sorteioCorInimigo.Next(listaCores.Length)];
            return cor;
        }
    }
}
